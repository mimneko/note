    \frac{\pi}{2} =

    \sum_{k=0}^{\infty} \frac{(2k)!}{2^{2k}(k!)^2} \frac{1}{2k+1} =

Sub ExportAsPNG()
    '
    ' スライド1~18の「西暦」テキストを変え、それぞれのスライドのオブジェクトを
    ' PowerPointファイルと同階層に18個のpng画像として出力します。
    '
    ' マクロが信用できない場合、また上手く動作しない場合は使用しなくても構いません。
    ' マクロを変更する際は、VBAをよく知っている方が行ってください。
    '
    
    
    Dim year As String
    Dim filePath As String
    Dim fileName As String
    Dim category(5) As String
    Dim shp As Shape
    Dim i As Long
    
    
    ' 西暦を指定
    ' https://learn.microsoft.com/ja-jp/office/vba/api/excel.application.inputbox
    year = InputBox("西暦を半角数字4ケタで入力してください。（例：1951）")
    If Len(year) <> 4 Then
        MsgBox "半角数字4ケタで入力してください。マクロを終了します。"
        Exit Sub
    End If
    
    
    ' 各分野
    category(0) = "Career_design"
    category(1) = "Learning_support"
    category(2) = "Sports"
    category(3) = "Library"
    category(4) = "Global"
    category(5) = "Volunteer"
    
    
    ' アクティブスライドのパスを取得
    ' https://cafe-mickey.com/hobby/programming/powerpoint-vba-pic/
    filePath = ActivePresentation.Path
    
        
    ' アクティブスライドの選択を解除
    ' https://okwave.jp/qa/q5585402.html
    With ActiveWindow.Selection
        If .Type <> ppSelectionNone Then
            .Unselect
        End If
    End With ' ActiveWindow.Selection
    
    
    For i = 1 To 18
        ' 西暦部分を、入力したものに変更
        ' https://qiita.com/taleau/items/47bf0c60a7ea593b74c5
        For Each shp In ActivePresentation.Slides(i).Shapes
            If shp.Name = "西暦" Then
                shp.TextFrame.TextRange.Text = year
            End If
        Next
        
        
        ' アクティブスライドの図形等をすべて選択
        ' https://learn.microsoft.com/ja-jp/office/vba/api/powerpoint.shapes.selectall
        'Dim myDocument
        'Set myDocument = ActivePresentation.Slides(1)
        'myDocument.Select
        'myDocument.Shapes.SelectAll
        With ActivePresentation.Slides(i)
            .Select
            .Shapes.SelectAll
        End With
        
        
        ' pngファイル名を決定
        fileName = "Open_Badge_" & year & "_" & category((i - 2) / 3) & "_" & ((i + 2) Mod 3 + 1)
        ' fileName = "sample"
        
        ' pngとして保存
        ' https://www.relief.jp/docs/powerpoint-vba-shape-save-as-picture.html
        ' https://my-tax-nology.com/powerpoint-vba-selected-shapes-saved-in-the-same-folder-as-current-file
        ActiveWindow.Selection.ShapeRange.Export _
            PathName:=filePath & "\" & fileName & ".png", _
            Filter:=ppShapeFormatPNG
    Next
    
End Sub
